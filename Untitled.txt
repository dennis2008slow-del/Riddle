let L = 0, R = 0, S = 0, time = 30, timerId = null, adCount = 0, paused = false, sdkReady = false;
let scores = JSON.parse(localStorage.getItem('rdScores') || '{}');

const $ = id => document.getElementById(id);
const toast = msg => { const t = $('toast'); t.textContent = msg; t.classList.add('show'); setTimeout(()=>t.classList.remove('show'), 2500); };

// Web Audio SFX
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
function beep(freq, dur) {
  const o = audioCtx.createOscillator(), g = audioCtx.createGain();
  o.connect(g); g.connect(audioCtx.destination);
  o.frequency.value = freq; o.type = 'sine';
  g.gain.setValueAtTime(0.3, audioCtx.currentTime);
  g.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + dur);
  o.start(); o.stop(audioCtx.currentTime + dur);
}

async function initSDK() {
  try {
    CrazyGames.SDK.game.loadingStart();
    await CrazyGames.SDK.init();
    sdkReady = true;
    // CrazyGames.SDK.ad.enableTestMode(); // Uncomment for GitHub testing only
    CrazyGames.SDK.game.loadingStop();
    CrazyGames.SDK.game.gameplayStart();
  } catch (e) { console.error('SDK init failed', e); }
}

function setBg() {
  document.body.style.background = LEVELS[L].bg;
  $('encourage').textContent = LEVELS[L].encourage;
}

function saveScore() {
  const today = new Date().toISOString().slice(0,10);
  if (!scores[today]) scores[today] = [];
  scores[today].push({level: L+1, score: S, time: Date.now()});
  scores[today].sort((a,b) => b.score - a.score || a.time - b.time);
  scores[today] = scores[today].slice(0,10);
  localStorage.setItem('rdScores', JSON.stringify(scores));
}

function getLeaderboard() {
  const today = new Date().toISOString().slice(0,10);
  return scores[today] || [];
}

function render() {
  setBg();
  if (R >= 10) { endLevel(); return; }
  const r = LEVELS[L].riddles[R];
  $('levelPill').textContent = `Level ${L+1}`;
  $('scorePill').textContent = `${S}/10`;
  $('riddleText').textContent = r.q;
  $('answer').value = '';
  $('feedback').textContent = '';
  $('feedback').className = '';
  $('hintBox').style.display = 'none';
  $('nextBtn').style.display = 'none';
  $('hintBtn').disabled = false;
  $('hintBtn').textContent = '💡 Hint (Watch Ad)';
  startTimer();
  $('answer').focus();
}

function startTimer() {
  clearInterval(timerId);
  time = 30;
  updateTimer();
  timerId = setInterval(() => {
    if (paused) return;
    time--;
    updateTimer();
    if (time <= 0) {
      clearInterval(timerId);
      $('feedback').textContent = 'Time up! Try again🥺';
      $('feedback').className = 'incorrect';
      beep(200, 0.3);
      setTimeout(next, 1500);
    }
  }, 1000);
}

function updateTimer() {
  $('timerPill').textContent = `${time}s`;
  if (time <= 10) $('timerPill').classList.add('pulse');
  else $('timerPill').classList.remove('pulse');
}

function check() {
  const val = $('answer').value.trim().toLowerCase();
  const ans = LEVELS[L].riddles[R].a.toLowerCase();
  if (val === ans) {
    $('feedback').textContent = 'Correct🤩';
    $('feedback').className = 'correct';
    S++;
    $('scorePill').textContent = `${S}/10`;
    clearInterval(timerId);
    $('nextBtn').style.display = 'block';
    $('checkBtn').disabled = true;
    beep(800, 0.15); setTimeout(()=>beep(1000, 0.15), 150);
  } else {
    $('feedback').textContent = 'Try again🥺';
    $('feedback').className = 'incorrect';
    beep(200, 0.3);
  }
}

async function watchAd(type) {
  if (!sdkReady) { toast('Ads only work on CrazyGames.com'); return; }
  $('hintBtn').disabled = true;
  $('timerPill').classList.add('pulse');
  paused = true;
  CrazyGames.SDK.game.gameplayStop();
  try {
    await CrazyGames.SDK.ad.requestAd('rewarded');
    if (type === 'hint') {
      const h = LEVELS[L].riddles[R].h;
      $('hintBox').textContent = h;
      $('hintBox').style.display = 'block';
      $('hintBtn').style.display = 'none';
      toast('Hint unlocked!');
    } else if (type === 'unlock') {
      adCount++;
      toast(`Ad watched! ${3-adCount} more to skip level`);
      if (adCount >= 3) { closeModal(); forceNextLevel(); }
    }
  } catch (e) {
    toast('Ad not completed. Try again.');
    $('hintBtn').disabled = false;
  } finally {
    paused = false;
    CrazyGames.SDK.game.gameplayStart();
  }
}

function next() {
  R++;
  $('checkBtn').disabled = false;
  render();
}

function endLevel() {
  clearInterval(timerId);
  CrazyGames.SDK.game.gameplayStop();
  saveScore();
  const passed = S >= 8;
  const canSkip = adCount >= 3;
  const lb = getLeaderboard();
  let html = `<h2>${passed? '🎉 Level Complete!' : '😔 Level Failed'}</h2>
    <div class="big">${passed? '✅' : '❌'}</div>
    <div class="stats"><div>Score: ${S}/10</div><div>${passed? 'You advance!' : 'Need 8/10 to advance'}</div><div>Ads watched: ${adCount}/3</div></div>`;
  if (L === 7 && passed) {
    html = `<h2>🏆 YOU BEAT RIDDLEDASH! 🏆</h2><div class="big">👑</div>
    <div class="stats"><div>Final Score: ${S}/10</div><div>Levels Cleared: 8/8</div><div>You are a Riddle Master!</div></div>`;
    CrazyGames.SDK.game.gameOver();
  }
  html += `<div class="leaderboard"><h3>Today's Top 10</h3><ol>`;
  if (lb.length === 0) html += `<li>No scores yet today</li>`;
  lb.forEach(e => { html += `<li>L${e.level}: ${e.score}/10</li>`; });
  html += `</ol></div><div>`;
  if (L < 7) {
    if (passed) html += `<button class="btn btn-next" onclick="nextLevel()">Continue to Level ${L+2}</button>`;
    else if (canSkip) html += `<button class="btn btn-next" onclick="forceNextLevel()">Skip to Level ${L+2}</button>`;
    else html += `<button class="btn btn-restart" onclick="retryLevel()">🔄 Restart Level</button><button class="btn btn-ad" onclick="watchAd('unlock')">📺 Watch Ad ${adCount}/3</button>`;
  } else {
    html += `<button class="btn btn-restart" onclick="restartGame()">🔄 Play Again</button>`;
  }
  html += `</div>`;
  $('scoreModal').innerHTML = html;
  $('scoreOverlay').classList.add('show');
}

function retryLevel() { closeModal(); R = 0; S = 0; adCount = 0; CrazyGames.SDK.game.gameplayStart(); render(); }
function restartGame() { closeModal(); L = 0; R = 0; S = 0; adCount = 0; CrazyGames.SDK.game.happyTime(); CrazyGames.SDK.game.gameplayStart(); render(); }
